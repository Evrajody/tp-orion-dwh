% Macro raccourcie : insère un diagramme avec son cadre.
%   #1 numéro  #2 titre  #3 sous-titre  #4 fichier (sans ext.)  #5 hauteur
\newcommand{\umldiag}[5]{%
  \begin{umlbox}{#1}{#2}{#3}
    \centering
    \includegraphics[height=#5,keepaspectratio,
                     max width=\linewidth]{build/#4}
  \end{umlbox}%
}
